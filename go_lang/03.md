# âœ… **A3. Reverse an array in-place**

```go
func main() {
    arr := [5]int{1,2,3,4,5}

    for i, j := 0, len(arr)-1; i < j; i, j = i+1, j-1 {
        arr[i], arr[j] = arr[j], arr[i]
    }

    fmt.Println(arr)
}
```

# âœ… **A5. Filter long strings**

```go
func filterLong(strs []string) []string {
    result := []string{}

    for _, s := range strs {
        if len(s) > 3 {
            result = append(result, s)
        }
    }
    return result
}

func main() {
    fmt.Println(filterLong([]string{"go", "java", "js", "rust", "ai"}))
}
```

# âœ… **A7. Student marks & topper**

```go
func main() {
    students := map[string]int{
        "Aman": 90,
        "Riya": 85,
        "Sahil": 92,
        "Tina": 88,
        "John": 75,
    }

    delete(students, "John")

    topper := ""
    topMarks := -1

    for name, marks := range students {
        if marks > topMarks {
            topMarks = marks
            topper = name
        }
    }

    println("Topper is:", topper)
}
```

# âœ… **A9. Swap keys & values**

```go
func swapMap(m map[int]string) map[string]int {
    result := make(map[string]int)

    for k, v := range m {
        result[v] = k
    }
    return result
}

func main() {
    m := map[int]string{1:"apple", 2:"banana"}
    fmt.Println(swapMap(m))
}
```

# âœ… **A11. Swap using pointers**

```go
func swap(a, b *int) {
    *a, *b = *b, *a
}

func main() {
    x, y := 10, 20
    swap(&x, &y)
    fmt.Println(x, y) // 20 10
}
```
